% --- SLIDES SOBRE OS ALGORITMOS DE AGRUPAMENTO (DENTRO DA METODOLOGIA) ---

\begin{frame}{KMeans}
\begin{columns}
    \begin{column}{0.55\textwidth}
        \begin{itemize}
            \item Algoritmo de agrupamento baseado em centroides.
            \item Inicializa $k$ centroides aleatórios e atribui cada ponto ao centro mais próximo.
            \item Atualiza os centroides como a média dos pontos atribuídos.
            \item Repete até convergência (centroides não mudam).
            \item Simples, eficiente, mas sensível a outliers e à escolha de $k$.
        \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
        \centering
        \includegraphics[width=0.8\textwidth]{../src/img/cmeans.jpg} \\[-0.5em]
        \scriptsize Exemplo de clusters do KMeans
    \end{column}
\end{columns}
\end{frame}

\begin{frame}{Fuzzy C-Means}
\begin{columns}
    \begin{column}{0.55\textwidth}
        \begin{itemize}
            \item Variante "fuzzy" do KMeans: cada ponto pertence a todos os clusters com um grau de pertinência.
            \item Minimiza uma função de custo ponderada pelas pertinências.
            \item Permite sobreposição entre clusters, útil para dados ambíguos.
            \item Mais flexível, mas requer ajuste do parâmetro de fuzzificação.
        \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
        \centering
        \includegraphics[width=0.8\textwidth]{../src/img/cmeans.jpg} \\[-0.5em]
        \scriptsize Exemplo de clusters do Fuzzy C-Means
    \end{column}
\end{columns}
\end{frame}

\begin{frame}{Agglomerativo}
\begin{columns}
    \begin{column}{0.55\textwidth}
        \begin{itemize}
            \item Algoritmo hierárquico: cada ponto começa em seu próprio cluster.
            \item Une os clusters mais próximos iterativamente até restar $k$ clusters.
            \item Não requer definição prévia de centroides.
            \item Gera uma árvore de agrupamento (dendrograma).
            \item Capaz de capturar estruturas complexas, mas pode ser computacionalmente custoso.
        \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
        \centering
        \includegraphics[width=0.8\textwidth]{../src/img/002-agglomerative-clustering-cut.jpg} \\[-0.5em]
        \scriptsize Exemplo de dendrograma (Agglomerativo)
    \end{column}
\end{columns}
\end{frame}

\begin{frame}{Bases de Dados}
\textbf{Adult:}
\begin{itemize}
    \item Informações demográficas e socioeconômicas
    \item Variáveis: idade, escolaridade, ocupação, sexo, raça, horas/semana, país de origem
    \item Variáveis categóricas codificadas numericamente
    \item Agrupamento sem uso da variável de saída (income)
\end{itemize}
\vspace{0.5em}
\textbf{Dry Bean:}
\begin{itemize}
    \item Características físicas/morfológicas de feijões
    \item Todas as variáveis numéricas
    \item Variável de saída (Class) usada apenas para avaliação
\end{itemize}
\end{frame}

\begin{frame}{Balanceamento das Classes}
\begin{itemize}
    \item Problema: distribuição desbalanceada entre as classes
    \item Solução: oversampling das classes minoritárias até igualar a maioria
    \item Balanceamento feito antes da divisão treino/teste
    \item Garante avaliação justa e sem viés
\end{itemize}
\end{frame}

\begin{frame}{Pipeline de Agrupamento}
\begin{enumerate}
    \item Balanceamento das classes
    \item Pré-processamento (normalização, codificação)
    \item Definição do número de clusters ($k$)
    \item Execução dos algoritmos
    \item Repetição dos experimentos (30 execuções, seeds diferentes)
    \item Avaliação quantitativa (acurácia média, desvio padrão, matriz de confusão)
    \item Visualização dos resultados (PCA, matrizes de confusão)
\end{enumerate}
\end{frame}

\begin{frame}{Métricas de Avaliação}
\begin{itemize}
    \item \textbf{Acurácia média:} proporção média de exemplos corretamente agrupados
    \item \textbf{Desvio padrão:} dispersão das acurácias nas repetições
    \item \textbf{Matriz de confusão:} mostra acertos e erros por classe
\end{itemize}
\vspace{0.5em}
\textbf{Matriz de confusão binária:}
\begin{center}
\begin{tabular}{|c|c|c|}
    \hline
    & \textbf{Predito: 1} & \textbf{Predito: 0} \\
    \hline
    \textbf{Real: 1} & VP & FN \\
    \hline
    \textbf{Real: 0} & FP & VN \\
    \hline
\end{tabular}
\end{center}
\begin{itemize}
    \item VP: verdadeiro positivo (classe real e predita são positivas)
    \item FP: falso positivo (classe real é negativa, mas predita como positiva)
    \item VN: verdadeiro negativo (classe real e predita são negativas)
    \item FN: falso negativo (classe real é positiva, mas predita como negativa)
\end{itemize}
\end{frame}

\section{Conclusão}

% --- CONCLUSÃO ---
\begin{frame}{Conclusão}
\begin{itemize}
    \item \textbf{Fuzzy C-Means} obteve o melhor desempenho geral, com acurácia média de 0,84 e baixo desvio padrão no Dry Bean, mostrando robustez para problemas multiclasse.
    \item \textbf{KMeans} apresentou desempenho intermediário: acurácia de 0,71 (Adult) e 0,75 (Dry Bean), com maior estabilidade no Adult (DP = 0,016).
    \item \textbf{Agglomerativo} teve o pior desempenho, especialmente no Dry Bean (acurácia média de 0,27 e DP = 0,15), indicando limitação para dados complexos ou com muitas classes.
    \item O desvio padrão elevado do Agglomerativo e do KMeans no Dry Bean evidencia instabilidade e sensibilidade à estrutura dos dados.
    \item A escolha do algoritmo deve considerar a natureza dos dados, número de classes e necessidade de robustez.
\end{itemize}
\end{frame}
